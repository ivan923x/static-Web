PACKAGE APP_CHART_PKG As
    /** 
    * 開發專案:  <a href="https://m1.im.cyut.edu.tw/ords/app111_apex1/r/wm-after-sales-service">木工機械售後服務系統</a> (PLDoc)<br/>

    /** 
    * <strong>取得近幾天的最早日期</strong>
    * 以秒為單位, 傳入欲查詢幾天前的天數, 將當下系統日期與之相減, 並回傳該日日期
    * 應用例子: 使用下拉式選單傳入參數, 限制統計的日期範圍
    * 回傳值為日期類型
    *
    * @example
    * APP_CHART_PKG.get_date_of_timeframe(:P7_TIMEFRAME); 
    *
    * @param p_seconds <em>Number</em> 秒數單位的天數
    * @returns <em>Date</em>
    */
    Function get_date_of_timeframe (
        p_seconds  IN NUMBER := 604800
    ) Return DATE;

    /** 
    * <strong>以年/月份查詢該月的第一天日期</strong>
    * 應用例子: 使用年份/月份下拉式選單傳入參數, 搭配函數 APP_CHART_PKG.get_last_date_of_month_picker() 使用, 限制統計的日期範圍
    * 回傳值為日期類型
    *
    * @example
    * APP_CHART_PKG.get_first_date_of_month_picker(:P7_DATE_PICKER_YEAR, :P7_DATE_PICKER_MONTH); 
    *
    * @param p_years <em>Varchar</em> 年份
    * @param p_month <em>Varchar</em> 月份
    * @returns <em>Date</em>
    */
    Function get_first_date_of_month_picker (
        p_years  IN VARCHAR2,
        p_month  IN VARCHAR2
    ) Return DATE;

    /** 
    * <strong>以年/月份查詢該月的最後天日期</strong>
    * 應用例子: 使用年份/月份下拉式選單傳入參數, 搭配函數 APP_CHART_PKG.get_first_date_of_month_picker() 使用, 限制統計的日期範圍
    * 回傳值為日期類型
    *
    * @example
    * APP_CHART_PKG.get_last_date_of_month_picker(:P7_DATE_PICKER_YEAR, :P7_DATE_PICKER_MONTH); 
    *
    * @param p_years <em>Varchar</em> 年份
    * @param p_month <em>Varchar</em> 月份
    * @returns <em>Date</em>
    */
    Function get_last_date_of_month_picker (
        p_years  IN VARCHAR2,
        p_month  IN VARCHAR2
    ) Return DATE;

    End APP_CHART_PKG;
